### 디버깅 예제 Ex_5_2_해결

```c
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#define SIZE 10

int main(void)
{
	int i, j;
	int level_a = 35, level_b = 67;
	int count_a = 0, count_b = 0, count_c = 0;
	int nums[10] = { 23, 34, 95, 68, 93, 36 };


	for (i = 0; i < SIZE; i++) {

		if (nums[i] <= 0) continue;
		if (nums[i] > 100) continue;
		if (nums[i] >= level_b)
			printf("nums[%d]= %d >= %d \n", i, nums[i],level_b),
			count_c++;
		else if (nums[i] >= level_a)
			printf("nums[%d]= %d >= %d \n", i, nums[i], level_a),
			count_b++;
		else if (nums[i] >= 0)
			printf("nums[%d]= %d >= 0 \n", i, nums[i]),
			count_a++;
	}

	printf("count_a = %d \n", count_a);
	printf("count_b = %d \n", count_b);
	printf("count_c = %d \n", count_c);
}
```

```c
	float y;
```
