#define _CRT_SECURE_NO_WARNINGS

#include <stdio.h>
#include <stdlib.h>

struct cmplx {
	float real;
	float imag;
};

struct cmplx multcmplx(struct cmplx x, struct cmplx y);

int main(void) {

	struct cmplx x, y, z;
	FILE    *infile;
	// FILE	*outfile;
	//char buffer[256];

	infile = fopen("indata.txt", "r");
	if (!infile) {
		printf("File open error: indata.txt");
		exit(1);
	}

	fscanf(infile, "%f, %f, %f, %f", &x.real, &x.imag, &y.real, &y.imag);
	printf("x= %.2f + %.2f * i\n", x.real, x.imag);
	printf("y= %.2f + %.2f * i\n", y.real, y.imag);

	z = multcmplx(x, y);
	printf("z = %.2f + %.2f * i\n", z.real, z.imag);

	fclose(infile);

	return 0;
}

struct cmplx multcmplx(struct cmplx x, struct cmplx y) {
	struct cmplx z;
	z.real = x.real * y.real - x.imag * y.imag;
	z.imag = x.real * y.imag + x.imag * y.real;
	return z;
}
