```c
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

void PrintString(int sel, FILE* fp);
void PrintAllString(FILE* fp);

int main() {

	int i;
	int sel;
	char str[100];
	FILE* fp;


	fp = fopen("string.txt", "w");
	if (fp == NULL) {
		printf("File open error: string.txt");
		return -1;
	}

	for (i = 0; i < 5; i++) {
		printf("%d 번째 문자열 입력:", i + 1);
		fgets(str, 100, stdin);
		fputs(str, fp);
	}


	fclose(fp);
	puts("완료");

	FILE* fp2;


	fp2 = fopen("string.txt", "r");
	if (fp2 == NULL) {
		printf("File open error: string.txt");
		return -1;
	}

	fputs("출력할 문자열 정보 입력 (6이상은 전체 출력) : ",stdout);
	scanf("%d", &sel);

	if (0<sel && sel <6)
		PrintString(sel,fp2);//값이 0보다 작거나 6보다 크면 전체 출력	
	else
		PrintAllString(fp2);

	fclose(fp2);
	puts("완료");


	return 0;

}

void PrintString(int sel, FILE* fp) {
	char str[100];

	for (int i = 0; i < sel - 1; i++) {
		fgets(str,100-1 , fp);
	}
	fgets(str, 100 - 1, fp);
	fput(str, stdout);

}

void PrintAllString(FILE* fp) {
	char str[100];

	while (1) {
		fgets(str, 100 - 1, fp);
		if (feof(fp) != 0) // 파일의 끝을 만나면 1을 반환
			break;

		fput(str, stdout);
	}
}
```

### while(getchar()!="\n") continue; 사용

```c

#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>

// 		while (getchar() != '\n') continue; 사용

int main() {

	int day;
	int ap;
	int ret;
	char str[100];
	
	FILE* fp = fopen("weather.dat","w");
	if (fp == NULL) {
		printf("안열림");
		return -1;
	}

	while (1) {
		fput("** 날씨 정보 입력 ** \n", stdout);

		// 1. 날짜 입력(숫자)
		fput("날짜 : ", stdout);
		scanf("%d", &day);
		if (day == -1) break;	// -1 입력 받으면 종료
		while (getchar() != "\n") continue;
		fwrite(&day, sizeof(int), 1, fp);

			
		// 2. 지역 입력(문자열)
		fputs("지역: ",stdout);
		fgets(str, 100, stdin);
		fputs(str, fp);

		// 3. 오전 / 오후 입력(문자)
		fput("오전(A)/오후(P) : ", stdout);
		fgetchar();
		fputchar();

		// 4. 날씨 정보 입력(문자열)
		fputs("날씨: ", stdout);
		fgets(str, 100, stdin);
		fputs(str, fp);
	}

	return 0;

}
```
