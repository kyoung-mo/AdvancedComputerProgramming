```c
#include <stdio.h>
#define _CRT_SECURE_NO_WARNINGS

void rectangle(int row, int col);

int main() {

	int row, col;
	
	printf("행의 크기를 입력하세요: ");
	scanf_s("%d", &row);


	printf("열의 크기를 입력하세요: ");
	scanf_s("%d", &col);

	rectangle(row, col);

	printf("row = %d, col =%d, 사각형의 크기는 %d * %d 입니다", row, col, row, col);
	return 0;
}

void rectangle(int row, int col) {
	int i, j;

	for (i = 0; i < row; i++) {
		for (j = 0; j < col; j++) {
			printf("*");
		}
		printf("\n");
	}

}
```

```c
#include <stdio.h>
#define _CRT_SECURE_NO_WARNINGS

void rectangleLine(int row, int col);

int main() {

	int row, col;
	
	printf("행의 크기를 입력하세요: ");
	scanf_s("%d", &row);


	printf("열의 크기를 입력하세요: ");
	scanf_s("%d", &col);

	rectangleLine(row, col);

	printf("row = %d, col =%d, 사각형의 크기는 %d * %d 입니다", row, col, row, col);
	return 0;
}

void rectangleLine(int row, int col) {
	int i, j;

	for (i = 0; i < row; i++) {
		for (j = 0; j < col; j++) {
			if (i == 0 || i == row-1)
				printf("*");
			else if (j == 0 || j == col - 1)
				printf("*");
			else
				printf(" ");
		}
		printf("\n");
	}

}
```

### Ex_2_0_6
```c
#include <stdio.h>
#define _CRT_SECURE_NO_WARNINGS


struct cmplx {
	float real;
	float imag;
};

struct cmplx cmplx_add(struct cmplx x, struct cmplx y);

int main() {

	struct cmplx x = { 2,3 }, y = { 4,5 }, z;
	printf("x=%.2f + i*%2f\n", x.real, x.imag);
	
	z = cmplx_add(x, y);
	printf("result is z = %.2f + i*%.2f\n", z.real, z.imag);

	return 0;
}


struct cmplx cmplx_add(struct cmplx x, struct cmplx y) {
	float real;
	float imag;

	struct cmplx z;

	real = x.real + y.real;
	imag = x.imag + y.imag;
	z.real = real;
	z.imag = imag;

	return z;

}
```
