	if (num >= 81)
		printf("A 학점 입니다. \n");
	else if (num >= 61)
		printf("B 학점 입니다.\n");
	else if (num >= 41)
		printf("C 학점 입니다. \n");
	else
		printf("D 학점 입니다.\n");
	return 0;

}
